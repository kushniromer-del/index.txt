function addToLb(cat, name, score) {
  let lb = getLb(cat);
  // seed fake data if empty
  if(lb.length < 5) {
    for(let i=0;i<8;i++) {
      lb.push({name:rand(FAKE_NAMES), score: Math.floor(Math.random()*800+100), country:rand(COUNTRIES)});
    }

  lb.push({name, score, country:'🇮🇱'});
  lb.sort((a,b)=>b.score-a.score);
  lb = lb.slice(0,20);
  saveLb(cat, lb);
  return lb.findIndex(e=>e.name===name&&e.score===score)+1;
}

function startSoloGame() {
  const name = el('solo-name').value.trim() || 'שחקן 1';
  const times = {easy:15, medium:10, hard:6};
  soloState = {
    name, score:0, level:1, streak:0, maxStreak:0,
    correct:0, questionsInLevel:0, timePerQ: times[selectedDiff],
    timeLeft:0, activeOpts:[], currentQ:null, multiPlayer:false,
    waitingForSkip:false
  };
  currentRule = RULES[0];
  show('screen-solo-game');
  nextQuestion();
  scheduleRuleChange();

function scheduleRuleChange() {
  clearTimeout(soloRuleTimer);
  const interval = soloState.level < 3 ? 30000 : soloState.level < 6 ? 20000 : 12000;
  soloRuleTimer = setTimeout(() => { if(el('screen-solo-game').classList.contains('active')) triggerRuleChange(); }, interval);
}

function triggerRuleChange() {
  const newRule = rand(RULES.filter(r=>r.id!==currentRule.id));
  currentRule = newRule;
  showRuleChangeOverlay(newRule.desc);
  scheduleRuleChange();
}

function showRuleChangeOverlay(text) {
  el('rule-change-text').textContent = text;
  el('rule-change-overlay').style.display='flex';
  setTimeout(()=>{ el('rule-change-overlay').style.display='none'; }, 2200);
}

function nextQuestion() {
  if(soloTimer) clearInterval(soloTimer);
  const q = getQuestion(soloState.level);
  soloState.currentQ = q;

  // Rule: SKIP
  if(currentRule.id === 'skip') {
    el('sg-question').textContent = '⏳ חכה — אל תלחץ!';
    el('sg-opts').innerHTML = '';
    el('sg-rule-banner').textContent = currentRule.label;
    soloState.waitingForSkip = true;
    startTimer(soloState.timePerQ, ()=>{
      soloState.waitingForSkip = false;
      soloState.score += 10 * soloState.level;
      soloState.streak++;
      if(soloState.streak > soloState.maxStreak) soloState.maxStreak = soloState.streak;
      soloState.correct++;
      updateSoloHud();
      progressLevel();
      nextQuestion();
    });
    return;
  }

  soloState.waitingForSkip = false;
  el('sg-question').textContent = q.q;
  el('sg-rule-banner').textContent = currentRule.label;

  let opts = shuffle(q.opts.map((o,i)=>({text:o, correct:i===q.a})));
  soloState.activeOpts = opts;
  el('sg-opts').innerHTML = opts.map((o,i)=>`
    <button class="q-opt" id="opt-${i}" onclick="handleSoloOpt(${i})">${o.text}</button>`).join('');

  startTimer(soloState.timePerQ, ()=>{ gameOver(false, 'timeout'); });
}

function handleSoloOpt(idx) {
  if(soloState.waitingForSkip) return;
  if(soloTimer) clearInterval(soloTimer);

  const opts = soloState.activeOpts;
  const chosen = opts[idx];
  let correct;

  if(currentRule.id === 'opposite') correct = !chosen.correct;
  else correct = chosen.correct;

  // Visual
  document.querySelectorAll('.q-opt').forEach(b=>b.disabled=true);
  const correctIdx = opts.findIndex(o=> currentRule.id==='opposite' ? !o.correct : o.correct);
  el('opt-'+correctIdx).classList.add('correct');
  if(!correct) el('opt-'+idx).classList.add('wrong');

  if(correct) {
    soloState.score += (10 + soloState.streak*2) * soloState.level;
    soloState.streak++;
    if(soloState.streak > soloState.maxStreak) soloState.maxStreak = soloState.streak;
    soloState.correct++;
    if(soloState.streak>=3) showCombo(soloState.streak);
    updateSoloHud();
    progressLevel();
    setTimeout(()=> nextQuestion(), 600);
  } else {
    el('sg-card').classList.add('flash-wrong');
    setTimeout(()=>{ gameOver(false,'wrong'); }, 600);
  }
}

function handleSkipClick() {
  if(!soloState.waitingForSkip) return;
  clearInterval(soloTimer);
  el('sg-card').classList.add('flash-wrong');
  setTimeout(()=>{ gameOver(false,'wrong'); }, 400);
}

function progressLevel() {
  soloState.questionsInLevel++;
  const needed = 3 + soloState.level;
  const pct = (soloState.questionsInLevel / needed)*100;
  el('sg-level-bar').style.width = Math.min(pct,100)+'%';
  el('sg-level-label').textContent = `${soloState.questionsInLevel}/${needed} לרמה הבאה`;
  if(soloState.questionsInLevel >= needed) {
    soloState.level++;
    soloState.questionsInLevel = 0;
    el('sg-level').textContent = soloState.level;
    triggerRuleChange();
  }
}

function updateSoloHud() {
  el('sg-score').textContent  = soloState.score;
  el('sg-level').textContent  = soloState.level;
  el('sg-streak').textContent = soloState.streak+'🔥';
}

function startTimer(sec, onDone) {
  soloState.timeLeft = sec;
  const bar = el('sg-timer-bar');
  bar.style.width = '100%';
  bar.classList.remove('warn');
  soloTimer = setInterval(()=>{
    soloState.timeLeft -= 0.1;
    const pct = (soloState.timeLeft/sec)*100;
    bar.style.width = pct+'%';
    if(pct < 30) bar.classList.add('warn');
    if(soloState.timeLeft <= 0) { clearInterval(soloTimer); onDone(); }
  }, 100);
}

function gameOver(isMulti, reason) {
  clearInterval(soloTimer);
  clearTimeout(soloRuleTimer);
  el('rule-change-overlay').style.display = 'none';

  if(isMulti) {
    multiState.scores[multiState.current] = soloState.score;
    if(multiState.current === 0) {
      multiState.current = 1;
      show('screen-solo-game');
      setTimeout(()=>startPlayerGame(1), 500);
    } else {
      showMultiResult();
    }
    return;
  }

  // Save to leaderboard
  const rank = addToLb('solo', soloState.name, soloState.score);
  const emojis = {timeout:'⏱',wrong:'💥'};
  const labels = {timeout:'אזל הזמן!',wrong:'טעות!'};

  el('go-emoji').textContent = emojis[reason]||'💥';
  el('go-name-label').textContent = soloState.name;
  el('go-score').textContent = soloState.score;
  el('go-level').textContent = soloState.level;
  el('go-streak').textContent = soloState.maxStreak;
  el('go-correct').textContent = soloState.correct;

  const rankColors = ['var(--yellow)','#C0C0C0','#CD7F32'];
  el('go-rank-badge').innerHTML = rank<=3
    ? `<span style="font-size:36px">${['🥇','🥈','🥉'][rank-1]}</span><br><span style="color:${rankColors[rank-1]};font-weight:900">מקום ${rank} בלידרבורד!</span>`
    : `<span style="color:rgba(240,240,255,0.6)">מקום ${rank} בלידרבורד</span>`;

  show('screen-gameover');
}

const TAP_OBJECTS = ['🐱','🐶','🐸','🦊','🐯','🦁','🐻','🐼','🦄','🐲'];
const TAP_RULES = [
  {should:true,  text:'🖐 תלחץ על ', cls:'tap'},
  {should:false, text:'🚫 אל תלחץ על ', cls:'notap'},
];
let tapState = {};
let tapRuleTimer, tapGameTimer, tapCountdown;

function startTapGame() {
  tapState = {score:0, streak:0, timeLeft:30, currentObj:'', currentRule:null, locked:false};
  show('screen-tap');
  nextTapRound();
  // countdown
  clearInterval(tapGameTimer);
  tapState.timeLeft=30;
  tapGameTimer = setInterval(()=>{
    tapState.timeLeft--;
    el('tap-timer').textContent = tapState.timeLeft;
    if(tapState.timeLeft<=0) { clearInterval(tapGameTimer); endMini('tap'); }
  },1000);
}

function nextTapRound() {
  tapState.locked = false;
  tapState.currentObj = rand(TAP_OBJECTS);
  tapState.currentRule = rand(TAP_RULES);
  el('tap-obj').textContent = tapState.currentObj;
  el('tap-obj').style.transform = '';
  el('tap-rule').textContent = tapState.currentRule.text + tapState.currentObj;
  el('tap-rule').className = 'tap-rule '+tapState.currentRule.cls;
  el('tap-score').textContent = tapState.score;
  el('tap-streak').textContent = tapState.streak+'🔥';
  clearTimeout(tapRuleTimer);
  const wait = 2500 - Math.min(tapState.score*5, 1500);
  tapRuleTimer = setTimeout(()=>{ if(!tapState.locked) { tapMissed(); } }, wait);
}

function handleTap() {
  if(tapState.locked) return;
  tapState.locked = true;
  clearTimeout(tapRuleTimer);
  if(tapState.currentRule.should) {
    tapState.score += 10 + tapState.streak;
    tapState.streak++;
    el('tap-obj').style.transform='scale(1.3)';
    setTimeout(()=>nextTapRound(),300);
  } else {
    tapState.streak=0;
    el('tap-obj').style.transform='scale(0.7)';
    setTimeout(()=>nextTapRound(),300);
  }
  el('tap-score').textContent = tapState.score;
  el('tap-streak').textContent = tapState.streak+'🔥';
}

function tapMissed() {
  if(tapState.currentRule.should) { tapState.streak=0; }
  else { tapState.score+=5; tapState.streak++; }
  el('tap-score').textContent = tapState.score;
  el('tap-streak').textContent = tapState.streak+'🔥';
  nextTapRound();
}

const MEM_ITEMS = ['🍎','🍊','🍋','🍇','🍓','🍔','🌮','🍕','🍦','🎂','⚽','🏀','🎸','🎹','🚗','✈️','🚀','🦋','🌸','💎'];
let memState = {};
let memTimer;

async function startMemoryGame() {
  memState = {score:0, round:1, showCount:3, phase:'show'};
  show('screen-memory');
  await sleep(300);
  memRound();
}

async function memRound() {
  el('mem-score').textContent = memState.score;
  el('mem-round').textContent = memState.round;
  const count = memState.showCount;
  const selected = shuffle(MEM_ITEMS).slice(0, count+2);
  const targets = selected.slice(0, count);
  memState.targets = targets;
  memState.allItems = shuffle(selected);
  memState.picked = [];

  // SHOW PHASE
  memState.phase='show';
  el('mem-phase').textContent='🧠 זוכר...';
  el('mem-instruction').textContent='';
  renderMemGrid(memState.allItems, true);
  await sleep(2000 + count*400);

  // HIDE PHASE
  memState.phase='pick';
  el('mem-phase').textContent='👆 איזה אובייקטים ראית?';
  el('mem-instruction').textContent=`בחר ${count} אובייקטים`;
  renderMemGrid(memState.allItems, false);
}

function renderMemGrid(items, visible) {
  el('mem-grid').innerHTML = items.map((item,i)=>`
    <div class="mem-card ${visible?'':'hidden'}" id="mc-${i}" onclick="pickMemCard(${i},'${item}')">${item}</div>
  `).join('');
}

function pickMemCard(idx, item) {
  if(memState.phase!=='pick') return;
  const card = el('mc-'+idx);
  if(card.classList.contains('selected')) { 
    card.classList.remove('selected'); card.textContent='';
    memState.picked = memState.picked.filter(x=>x!==item);
    return;
  }
  card.classList.add('selected');
  card.textContent=item;
  memState.picked.push(item);
  if(memState.picked.length === memState.targets.length) checkMemory();
}

async function checkMemory() {
  memState.phase='result';
  let correct=0;
  memState.allItems.forEach((item,i)=>{
    const card=el('mc-'+i);
    if(memState.targets.includes(item)) {
      card.classList.add('correct'); card.textContent=item;
      if(memState.picked.includes(item)) correct++;
    } else if(card.classList.contains('selected')) {
      card.classList.remove('selected');
      card.classList.add('wrong');
    }
  });

  const allRight = correct===memState.targets.length;
  el('mem-phase').textContent = allRight?'✅ מעולה!':'❌ נסה שוב!';

  if(allRight) {
    memState.score += 20 * memState.round;
    memState.round++;
    memState.showCount = Math.min(3+Math.floor(memState.round/2), 8);
    el('mem-score').textContent = memState.score;
    await sleep(1500);
    memRound();
  } else {
    await sleep(1500);
    endMini('memory');
  }
}

const ODD_SETS = [
  {items:['🍎','🍊','🍋','🍇','🚗'], label:'פירות', odd:4},
  {items:['🐱','🐶','🐸','🚀','🦊'], label:'חיות', odd:3},
  {items:['⚽','🏀','🎾','🏐','🎸'], label:'כדורים', odd:4},
  {items:['🌹','🌺','🌻','🌸','🍕'], label:'פרחים', odd:4},
  {items:['🚗','✈️','🚀','🚂','🐱'], label:'כלי תחבורה', odd:4},
  {items:['1️⃣','2️⃣','3️⃣','🔴','4️⃣'], label:'מספרים', odd:3},
  {items:['🎵','🎶','🎸','🎹','🍔'], label:'מוזיקה', odd:4},
  {items:['🔴','🟠','🟡','🟢','🦁'], label:'עיגולים צבעוניים', odd:4},
  {items:['A','B','C','D','3'], label:'אותיות', odd:4},
  {items:['😀','😂','😍','😎','🏠'], label:'אמוג׳י פנים', odd:4},
];

let oddState = {};
let oddTimer;

function startOddGame() {
  oddState = {score:0, level:1, timePerQ:10, usedSets:[]};
  show('screen-odd');
  nextOddRound();
}

function nextOddRound() {
  if(oddTimer) clearInterval(oddTimer);
  el('odd-score').textContent = oddState.score;
  el('odd-level').textContent = oddState.level;

  // Pick unused set
  let available = ODD_SETS.filter((_,i)=>!oddState.usedSets.includes(i));
  if(!available.length) { oddState.usedSets=[]; available=ODD_SETS; }
  const setIdx = ODD_SETS.indexOf(rand(available));
  oddState.usedSets.push(setIdx);
  const set = ODD_SETS[setIdx];

  el('odd-question').textContent = `🔍 מצא מה לא שייך ל: ${set.label}`;
  const shuffledItems = shuffle(set.items.map((item,i)=>({item,isOdd:i===set.odd})));
  oddState.currentOdd = shuffledItems.findIndex(x=>x.isOdd);

  el('odd-grid').innerHTML = shuffledItems.map((x,i)=>`
    <button class="odd-card" id="odd-${i}" onclick="pickOdd(${i},${x.isOdd})">${x.item}</button>
  `).join('');

  // Timer
  oddState.timeLeft = oddState.timePerQ;
  el('odd-timer').textContent = oddState.timeLeft;
  el('odd-timer-bar').style.background='var(--yellow)';
  el('odd-timer-bar').style.width='100%';

  oddTimer = setInterval(()=>{
    oddState.timeLeft--;
    el('odd-timer').textContent = oddState.timeLeft;
    const pct = (oddState.timeLeft/oddState.timePerQ)*100;
    el('odd-timer-bar').style.width=pct+'%';
    if(pct<30) el('odd-timer-bar').style.background='var(--pink)';
    if(oddState.timeLeft<=0) { clearInterval(oddTimer); endMini('odd'); }
  },1000);
}

function pickOdd(idx, isOdd) {
  clearInterval(oddTimer);
  const card = el('odd-'+idx);
  if(isOdd) {
    card.classList.add('correct');
    const bonus = Math.max(1, oddState.timeLeft);
    oddState.score += 10 * oddState.level + bonus;
    oddState.level++;
    oddState.timePerQ = Math.max(4, 10 - Math.floor(oddState.level/2));
    el('odd-score').textContent = oddState.score;
    setTimeout(()=>nextOddRound(), 700);
  } else {
    card.classList.add('wrong');
    const correctCard = el('odd-'+oddState.currentOdd);
    if(correctCard) correctCard.classList.add('correct');
    setTimeout(()=>endMini('odd'), 900);
  }
}

function stopMini(type) { clearInterval(tapGameTimer); clearTimeout(tapRuleTimer); clearInterval(oddTimer); show('screen-home'); }

function endMini(type) {
  clearInterval(tapGameTimer); clearTimeout(tapRuleTimer); clearInterval(oddTimer);
  let score, emoji, title, retry;
  if(type==='tap') {
    score=tapState.score; emoji='🐱'; title='לחץ / אל תלחץ';
    retry=startTapGame;
  } else if(type==='memory') {
    score=memState.score; emoji='🧩'; title='זיכרון מהיר';
    retry=startMemoryGame;
  } else {
    score=oddState.score; emoji='🎯'; title='מצא חריג';
    retry=startOddGame;
  }

  const rank = addToLb(type, 'אתה', score);
  el('mini-over-title').textContent=title;
  el('mini-over-emoji').textContent=emoji;
  el('mini-over-score').textContent=score;
  el('mini-over-rank').textContent = rank<=3 ? ['🥇 מקום 1!','🥈 מקום 2!','🥉 מקום 3!'][rank-1] : `מקום ${rank} בלידרבורד`;
  el('mini-retry-btn').onclick = retry;
  show('screen-mini-over');
}

    navigator.clipboard.writeText(text).then(()=>alert('הועתק ללוח! 📋'));
  }
}
